cmake_minimum_required(VERSION 2.6)

ENABLE_TESTING()

SET (M_NAME samtools)
SET (INTERFACE python_${M_NAME}.i)

IF(NOT BUILD_LIBS)
  SET (MAP_ROOT ../../../../..)
  SET (CMAKE_MODULE_PATH ${MAP_ROOT}/tools/cmake-support/modules)
ENDIF(NOT BUILD_LIBS)

SET(USE_C_LINKAGE TRUE)
SET(SAMTOOLS_PATH ${MAP_ROOT}/contrib/samtools)
SET(USE_SWIG_INCLUDEPATH ${SAMTOOLS_PATH})
INCLUDE_DIRECTORIES(${SAMTOOLS_PATH})

SET(CMAKE_LIBRARY_OUTPUT_DIRECTORY biolib)

FIND_PACKAGE(MapPython REQUIRED)

POST_BUILD_PYTHON_BINDINGS()
TEST_PYTHON_BINDINGS()
INSTALL_PYTHON_BINDINGS()
