cmake_minimum_required(VERSION 2.6)

ENABLE_TESTING()

SET (M_NAME emboss)
SET (M_VERSION 6.1.0)
SET (INTERFACE python_${M_NAME}.i)

IF(NOT BUILD_LIBS)
  SET (MAP_ROOT ../../../../..)
  SET (CMAKE_MODULE_PATH ${MAP_ROOT}/tools/cmake-support/modules)
ENDIF(NOT BUILD_LIBS)

SET(CMAKE_SWIG_FLAGS "-DSWIG_PYTHON_SILENT_MEMLEAK")
add_definitions(-DSWIG_PYTHON_SILENT_MEMLEAK)

SET(USE_C_LINKAGE TRUE)
SET(EMBOSS_PATH ${MAP_ROOT}/contrib/EMBOSS-${M_VERSION})
SET(USE_SWIG_INCLUDEPATH ${EMBOSS_PATH}/ajax)

INCLUDE_DIRECTORIES(${EMBOSS_PATH}/ajax)

SET(CMAKE_LIBRARY_OUTPUT_DIRECTORY biolib)

FIND_PACKAGE(MapPython REQUIRED)

POST_BUILD_PYTHON_BINDINGS()
TEST_PYTHON_BINDINGS()
INSTALL_PYTHON_BINDINGS()
